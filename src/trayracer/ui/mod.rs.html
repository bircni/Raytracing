26/// Main application
27/// This holds all the UI elements and application state
28pub struct App {
29    current_tab: Tab,
30    render: Render,
31    properties: Properties,
32    statusbar: StatusBar,
33    preview: Preview,
34    render_result: RenderResult,
35    yaml_menu: YamlMenu,
36    scene: Arc<RwLock<Option<Scene>>>,
37}
38
39#[derive(PartialEq)]
40enum Tab {
41    Preview,
42    RenderResult,
43}
44
45impl App {
46    pub fn new(cc: &CreationContext<'_>) -> anyhow::Result<Self> {
47        egui_extras::install_image_loaders(&cc.egui_ctx);
48
49        // Initialize the preview renderer with the wgpu context
50        preview::gpu::init_wgpu(
51            cc.wgpu_render_state
52                .as_ref()
53                .context("Failed to get wgpu context")?,
54        );
55
56        // create initial render texture (GPU exclusive) and image buffer (CPU exclusive)
57        let render_texture = cc.egui_ctx.load_texture(
58            "render",
59            ImageData::Color(Arc::new(ColorImage::example())),
60            TextureOptions::default(),
61        );
62        let image_buffer = Arc::new(Mutex::new(ImageBuffer::new(0, 0)));
63
64        cc.egui_ctx.style_mut(|s| {
65            s.text_styles.insert(
66                TextStyle::Name("subheading".into()),
67                TextStyle::Monospace.resolve(s),
68            );
69            s.text_styles
70                .insert(TextStyle::Body, TextStyle::Monospace.resolve(s));
71            s.spacing.item_spacing = vec2(10.0, f32::consts::PI * 1.76643);
72        });
73
74        let scene = Arc::new(RwLock::new(None));
75
76        Ok(Self {
77            current_tab: Tab::Preview,
78            render: Render::new(render_texture, image_buffer),
79            properties: Properties::new(),
80            statusbar: StatusBar::new(),
81            preview: Preview::new(Arc::<RwLock<Option<Scene>>>::clone(&scene)),
82            render_result: RenderResult::new(),
83            yaml_menu: YamlMenu::new(),
84            scene,
85        })
86    }
87}
88
89/// Main application loop (called every frame)
90impl eframe::App for App {
91    fn update(&mut self, ctx: &egui::Context, _frame: &mut eframe::Frame) {
92        // check if the render thread has finished and reset it
93        self.render
94            .thread
95            .as_ref()
96            .is_some_and(JoinHandle::is_finished)
97            .then(|| {
98                self.render.thread = None;
99                self.render.cancel.store(false, Ordering::Relaxed);
100            });
101
102        // lock the scene for the duration of the frame
103        let mut scene = self.scene.write();
104        CentralPanel::default().show(ctx, |ui| {
105            self.statusbar
106                .show(ui, scene.as_mut(), &mut self.render, &mut self.current_tab);
107
108            ui.vertical_centered(|ui| {
109                ui.separator();
110            });
111
112            match self.current_tab {
113                Tab::Preview => {
114                    SidePanel::right("panel")
115                        .show_separator_line(true)
116                        .show_inside(ui, |ui| {
117                            ScrollArea::new([false, true]).show(ui, |ui| {
118                                self.yaml_menu.show(&mut scene, ui);
119
120                                ui.separator();
121
122                                if let Some(scene) = scene.as_mut() {
123                                    self.properties.show(scene, ui, &self.render);
124                                }
125                            });
126                        });
127
128                    //if let Some(scene) = scene.as_mut() {
129                    //    self.preview.show(ui, scene);
130                    //} else {
131                    //    ui.with_layout(Layout::left_to_right(Align::Center), |ui| {
132                    //        ui.horizontal(|ui| {
133                    //            ui.vertical_centered(|ui| {
134                    //                ui.heading(t!("no_scene_loaded"));
135                    //                ui.label(RichText::new(t!("drop_yaml")));
136                    //            });
137                    //        });
138                    //    });
139                    //}
140                    self.preview.show(ui, &mut scene);
141                }
142                Tab::RenderResult => {
143                    if let Some(scene) = scene.as_ref() {
144                        self.render_result.show(ui, scene, &self.render);
145                    }
146                }
147            }
148        });
149    }
