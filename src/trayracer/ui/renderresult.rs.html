1use crate::{raytracer::render::Render, scene::Scene};
2use egui::{Color32, CornerRadius, CursorIcon, Frame, Rect, Sense, Stroke, Ui, Vec2, pos2};
3
4pub struct RenderResult {
5    // zoom factor where 0 is no zoom
6    zoom: f32,
7    position: Vec2,
8}
9
10impl RenderResult {
11    pub const fn new() -> Self {
12        Self {
13            zoom: 0.0,
14            position: Vec2::ZERO,
15        }
16    }
17
18    pub fn show(&mut self, ui: &mut Ui, scene: &Scene, render: &Render) {
19        Frame::canvas(ui.style()).outer_margin(10.0).show(ui, |ui| {
20            let (response, painter) = ui.allocate_painter(ui.available_size(), Sense::drag());
21
22            let response = response.on_hover_and_drag_cursor(CursorIcon::Grab);
23
24            // Check if the dialog is being hovered over or active
25            if response.has_focus() || response.hovered() {
26                self.zoom += ui.input(|i| i.raw_scroll_delta.y);
27                self.zoom = self.zoom.clamp(
28                    -response.rect.width().min(response.rect.height()) / 4.0,
29                    f32::INFINITY,
30                );
31                self.position += response.drag_delta();
32            }
33
34            response.double_clicked().then(|| {
35                self.zoom = 0.0;
36                self.position = Vec2::ZERO;
37            });
38
39            // paint gray grid
40            let cell_size = 25.0;
41            for y in 0..=response.rect.height() as usize / cell_size as usize {
42                for x in 0..=response.rect.width() as usize / cell_size as usize {
43                    painter.rect(
44                        Rect::from_min_size(
45                            pos2(
46                                (x as f32).mul_add(cell_size, response.rect.left()),
47                                (y as f32).mul_add(cell_size, response.rect.top()),
48                            ),
49                            Vec2::splat(cell_size),
50                        ),
51                        CornerRadius::default(),
52                        if (x + y) % 2 == 0 {
53                            Color32::GRAY
54                        } else {
55                            Color32::DARK_GRAY
56                        },
57                        Stroke::NONE,
58                        egui::StrokeKind::Outside,
59                    );
60                }
61            }
62
63            let render_aspect = scene.camera.resolution.0 as f32 / scene.camera.resolution.1 as f32;
64            let rect = Rect::from_min_size(
65                response.rect.min,
66                // keep aspect ratio
67                Vec2::new(
68                    response
69                        .rect
70                        .width()
71                        .min(response.rect.height() * render_aspect),
72                    response
73                        .rect
74                        .height()
75                        .min(response.rect.width() / render_aspect),
76                ),
77            );
78
79            // center rect
80            let rect = Rect::from_min_size(
81                rect.min + (response.rect.size() - rect.size()) / 2.0,
82                rect.size(),
83            );
84
85            painter.image(
86                render.texture.id(),
87                rect.translate(self.position)
88                    .expand2(Vec2::new(self.zoom * render_aspect, self.zoom)),
89                Rect::from_min_max(pos2(0.0, 0.0), pos2(1.0, 1.0)),
90                Color32::WHITE,
91            );
92        });
93    }
94}
