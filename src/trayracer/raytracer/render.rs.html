15pub struct Render {
16    pub texture: TextureHandle,
17    /// Progress of the rendering in the range [0, `u16::MAX`]
18    pub progress: Arc<AtomicU16>,
19    pub thread: Option<thread::JoinHandle<()>>,
20    /// Cancel the rendering if true
21    pub cancel: Arc<AtomicBool>,
22    pub image: Arc<Mutex<RgbImage>>,
23    /// Write the rendering time in milliseconds
24    pub time: Arc<AtomicU32>,
25}
26
27impl Render {
28    pub fn new(texture: TextureHandle, image: Arc<Mutex<RgbImage>>) -> Self {
29        Self {
30            texture,
31            progress: Arc::new(AtomicU16::new(0)),
32            thread: None,
33            cancel: Arc::new(AtomicBool::new(false)),
34            image,
35            time: Arc::new(AtomicU32::new(0)),
36        }
37    }
38
39    pub fn render(&mut self, ctx: egui::Context, scene: &Scene) {
40        let rsize = scene.camera.resolution;
41        info!("Rendering scene with resolution {rsize:?}");
42
43        // resize texture and image buffer to match the new resolution
44        self.texture.set(
45            ImageData::Color(Arc::new(ColorImage {
46                size: [rsize.0 as usize, rsize.1 as usize],
47                pixels: vec![Color32::BLACK; (rsize.0 * rsize.1) as usize],
48            })),
49            TextureOptions::default(),
50        );
51        *self.image.lock() = RgbImage::new(rsize.0, rsize.1);
52
53        // reset progress and time
54        self.progress.store(0, Ordering::Relaxed);
55        self.time.store(0, Ordering::Relaxed);
56
57        let args = RenderingThread {
58            cancel: Arc::<AtomicBool>::clone(&self.cancel),
59            ctx,
60            scene: scene.clone(),
61            progress: Arc::<AtomicU16>::clone(&self.progress),
62            texture: self.texture.clone(),
63            image: Arc::<Mutex<RgbImage>>::clone(&self.image),
64            time: Arc::<AtomicU32>::clone(&self.time),
65        };
66
67        // spawn rendering thread
68        self.thread = Some(thread::spawn(move || {
69            args.run();
70        }));
71    }
72}
73
74struct RenderingThread {
75    ctx: egui::Context,
76    scene: Scene,
77    /// egui Texture (GPU exclusive)
78    texture: TextureHandle,
79    /// image data (CPU exclusive)
80    image: Arc<Mutex<RgbImage>>,
81    /// Cancel the rendering if true
82    cancel: Arc<AtomicBool>,
83    /// Progress of the rendering in the range [0, `u16::MAX`]
84    progress: Arc<AtomicU16>,
85    /// Write the rendering time in milliseconds
86    time: Arc<AtomicU32>,
87}
88
89impl RenderingThread {
90    #[expect(
91        clippy::significant_drop_tightening,
92        reason = "no need to drop the texture"
93    )]
94    /// main rendering thread
95    fn run(self) {
96        let start = Instant::now();
97
98        let (width, height) = self.image.lock().dimensions();
99
100        // TODO: make block size adaptive to the resolution
101        // this will currently cause unrendered pixels if
102        // the resolution is not a multiple of 20
103        let block_size = [width / 20, height / 20];
104        let anti_aliasing = self.scene.settings.anti_aliasing;
105        let raytracer = Raytracer::new(self.scene, 1e-5, 5);
106
107        let blocks_rendered = AtomicUsize::new(0);
108
109        (0..height / block_size[1])
110            .flat_map(|y_block| (0..width / block_size[0]).map(move |x_block| (x_block, y_block)))
111            // parallelize iterator over blocks
112            .par_bridge()
113            .take_any_while(|_| !self.cancel.load(Ordering::Relaxed))
114            .map(|(x_block, y_block)| {
115                debug!(
116                    "rendering block ({}, {}) of ({}, {}) ({:.2}%)",
117                    x_block,
118                    y_block,
119                    width / block_size[0],
120                    height / block_size[1],
121                    (x_block + y_block * width / block_size[0]) as f32
122                        / (width / block_size[0] * height / block_size[1]) as f32
123                        * 100.0
124                );
125
126                let pixels = (0..block_size[0] * block_size[1])
127                    // parallelize over pixels
128                    .into_par_iter()
129                    .map(|i| {
130                        let x = i % block_size[0] + x_block * block_size[0];
131                        let y = i / block_size[0] + y_block * block_size[1];
132                        raytracer.render((x, y), (width, height), anti_aliasing)
133                    })
134                    .map(|c| {
135                        Color32::from_rgb(
136                            (c.x * 255.0) as u8,
137                            (c.y * 255.0) as u8,
138                            (c.z * 255.0) as u8,
139                        )
140                    })
141                    .collect::<Vec<_>>();
142
143                self.progress.store(
144                    ((blocks_rendered.fetch_add(1, Ordering::Relaxed) as f32)
145                        / (width / block_size[0] * height / block_size[1]) as f32
146                        * f32::from(u16::MAX))
147                    .round() as u16,
148                    Ordering::Relaxed,
149                );
150
151                (pixels, x_block, y_block)
152            })
153            // take while not cancelled
154            .take_any_while(|_| !self.cancel.load(Ordering::Relaxed))
155            .for_each_with(self.texture, |texture, (pixels, x_block, y_block)| {
156                // copy pixels to texture
157                texture.set_partial(
158                    [
159                        (x_block * block_size[0]) as usize,
160                        (y_block * block_size[1]) as usize,
161                    ],
162                    ImageData::Color(Arc::new(ColorImage {
163                        size: [block_size[0] as usize, block_size[1] as usize],
164                        pixels: pixels.clone(),
165                    })),
166                    TextureOptions::default(),
167                );
168
169                // copy pixels to image
170                let mut image = self.image.lock();
171                for x in 0..block_size[0] {
172                    for y in 0..block_size[1] {
173                        image.put_pixel(
174                            x_block * block_size[0] + x,
175                            y_block * block_size[1] + y,
176                            image::Rgb([
177                                pixels[(x + y * block_size[0]) as usize].r(),
178                                pixels[(x + y * block_size[0]) as usize].g(),
179                                pixels[(x + y * block_size[0]) as usize].b(),
180                            ]),
181                        );
182                    }
183                }
184
185                self.ctx.request_repaint();
186            });
187
188        self.progress.store(u16::MAX, Ordering::Relaxed);
189        self.time
190            .store(start.elapsed().as_millis() as u32, Ordering::Relaxed);
191
192        info!("rendering finished: {:?}", start.elapsed());
193    }
