34impl Raytracer {
35    const NO_MATERIAL_COLOR: Color = Color::new(0.9, 0.9, 0.9);
36
37    pub const fn new(scene: Scene, delta: f32, max_depth: u32) -> Self {
38        Self {
39            scene,
40            delta,
41            max_depth,
42        }
43    }
44
45    fn raycast(&self, ray: Ray) -> Option<Hit<'_>> {
46        self.scene
47            .objects
48            .iter()
49            .filter_map(|o| o.intersect(ray, self.delta))
50            .min_by_key(|h| OrderedFloat((h.point - ray.origin).norm()))
51    }
52
53    fn reflect(incoming: Vector3<f32>, normal: Vector3<f32>) -> Vector3<f32> {
54        incoming - 2.0 * incoming.dot(&normal) * normal
55    }
56
57    fn skybox(&self, direction: Vector3<f32>) -> Color {
58        match &self.scene.settings.skybox {
59            Skybox::Image { image, .. } => {
60                let direction = direction
61                    .try_normalize(f32::EPSILON)
62                    .unwrap_or_else(Vector3::y);
63
64                // spherical mapping
65                let x = ((0.5 + direction.z.atan2(direction.x) / (2.0 * f32::consts::PI))
66                    * image.width() as f32) as u32
67                    % image.width();
68                let y = ((0.5 - direction.y.asin() / f32::consts::PI) * image.height() as f32)
69                    as u32
70                    % image.height();
71
72                let pixel = image.get_pixel(x, y);
73
74                Color::new(
75                    f32::from(pixel[0]) / 255.0,
76                    f32::from(pixel[1]) / 255.0,
77                    f32::from(pixel[2]) / 255.0,
78                )
79            }
80            Skybox::Color(color) => *color,
81        }
82    }
83
84    fn texture(texture: &RgbImage, uv: Vector2<f32>) -> Color {
85        let x = (uv.x * texture.width() as f32) as u32 % texture.width();
86        let y = ((1.0 - uv.y) * texture.height() as f32) as u32 % texture.height();
87        let pixel = texture.get_pixel(x, y);
88        Color::new(
89            f32::from(pixel[0]) / 255.0,
90            f32::from(pixel[1]) / 255.0,
91            f32::from(pixel[2]) / 255.0,
92        )
93    }

95    /// Raycast and continue on hits if the material is transparent
96    fn raycast_transparent(&self, ray: Ray) -> Box<[Hit<'_>]> {
97        let mut hits = Vec::<Hit<'_>>::new();
98        let mut ray = ray;
99
100        while let Some(hit) = self.raycast(ray) {
101            hits.push(hit.clone());
102
103            if let Some(material) = hit.material {
104                if material.illumination_model.transparency() {
105                    // hochwissenschaftliche Formel
106                    ray.origin += ray.direction * 0.05;
107                    continue;
108                }
109            }
110            break;
111        }
112
113        hits.into_boxed_slice()
114    }
115
116    fn shade(&self, ray: Ray, depth: u32) -> Color {
117        // hochwissnschaftliche Formel +- x
118        self.raycast_transparent(ray).last().map_or_else(
119            || self.skybox(ray.direction),
120            |hit| self.shade_impl(ray, hit, depth),
121        )
122    }
123
124    fn shade_impl(&self, ray: Ray, hit: &Hit<'_>, depth: u32) -> Color {
125        if depth >= self.max_depth {
126            return self.skybox(ray.direction);
127        }
128
129        let diffuse_color = hit
130            .material
131            .and_then(|m| m.diffuse_texture.as_ref())
132            .map(|map| Self::texture(map, hit.uv))
133            .or_else(|| hit.material.and_then(|m| m.diffuse_color))
134            .unwrap_or(Self::NO_MATERIAL_COLOR);
135
136        let specular_color = hit
137            .material
138            .and_then(|m| m.specular_color)
139            .map_or(Self::NO_MATERIAL_COLOR, Color::from);
140
141        let mut color = self
142            .scene
143            .settings
144            .ambient_color
145            .component_mul(&diffuse_color)
146            * self.scene.settings.ambient_intensity;
147
148        for light in &self.scene.lights {
149            let light_direction = (light.position - hit.point).normalize();
150            let light_ray = Ray {
151                origin: hit.point + light_direction * self.delta,
152                direction: light_direction,
153            };
154
155            let light_transmission_color = self
156                .raycast_transparent(light_ray)
157                .iter()
158                .last()
159                .map_or_else(
160                    || Color::from_element(1.0),
161                    |hit| {
162                        color.component_mul(
163                            &hit.material
164                                .and_then(|m| m.diffuse_color)
165                                .unwrap_or_else(|| Color::from_element(1.0)),
166                        ) * hit.material.and_then(|m| m.dissolve).unwrap_or(1.0)
167                    },
168                )
169                .component_mul(&light.color);
170
171            if light_transmission_color.norm() < 0.01 {
172                continue;
173            }
174
175            // diffuse component
176            let light_intensity = light.intensity / (light.position - hit.point).norm_squared();
177            let diffuse_intensity = light_direction.dot(&hit.normal).max(0.0) * light_intensity;
178            color += diffuse_color.component_mul(&light_transmission_color) * diffuse_intensity;
179
180            // specular component
181            if hit
182                .material
183                .is_some_and(|m| m.illumination_model.specular())
184            {
185                let specular_intensity = light_direction
186                    .dot(&Self::reflect(-ray.direction, hit.normal))
187                    .max(0.0)
188                    .powf(
189                        hit.material
190                            .and_then(|m| m.specular_exponent)
191                            .unwrap_or(1.0),
192                    )
193                    * light_intensity;
194                color +=
195                    specular_color.component_mul(&light_transmission_color) * specular_intensity;
196            }
197
198            // reflection
199            if hit
200                .material
201                .is_some_and(|m| m.illumination_model.reflection())
202            {
203                let reflection_ray = Ray {
204                    origin: hit.point + hit.normal * self.delta,
205                    direction: Self::reflect(ray.direction, hit.normal),
206                };
207                color += color.component_mul(&self.shade(reflection_ray, depth + 1));
208            }
209        }
210
211        color
212    }

214    /// Render a pixel at the given coordinates.
215    /// x and y are in the range 0..width and 0..height
216    /// where (0, 0) is the top left corner.
217    ///Anti-aliasing is done by sampling multiple rays per pixel, enhanced with stratified sampling.
218    pub fn render(
219        &self,
220        (x, y): (u32, u32),
221        (width, height): (u32, u32),
222        anti_aliasing: bool,
223    ) -> Color {
224        if anti_aliasing {
225            let samples_per_pixel = self.scene.settings.samples;
226            let sqrt_samples = (samples_per_pixel as f32).sqrt() as u32;
227
228            (0..samples_per_pixel)
229                .into_par_iter()
230                .map(|i| {
231                    let xi = i % sqrt_samples;
232                    let yi = i / sqrt_samples;
233                    let jitter_x = (x as f32
234                        + (xi as f32 + rand::random::<f32>().mul_add(2.0, -1.0))
235                            / sqrt_samples as f32)
236                        / width as f32;
237                    let jitter_y = (y as f32
238                        + (yi as f32 + rand::random::<f32>().mul_add(2.0, -1.0))
239                            / sqrt_samples as f32)
240                        / height as f32;
241                    let x = jitter_x.mul_add(2.0, -1.0) * (width as f32 / height as f32);
242                    let y = jitter_y.mul_add(2.0, -1.0);
243                    let ray = self.scene.camera.ray(x, y);
244
245                    self.raycast(ray)
246                        .map_or_else(|| self.skybox(ray.direction), |_hit| self.shade(ray, 0))
247                })
248                .sum::<Color>()
249                / samples_per_pixel as f32
250        } else {
251            let x = (x as f32 / width as f32).mul_add(2.0, -1.0) * (width as f32 / height as f32);
252            let y = (y as f32 / height as f32).mul_add(2.0, -1.0);
253
254            let ray = self.scene.camera.ray(x, y);
255            self.shade(ray, 0)
256        }
257    }
